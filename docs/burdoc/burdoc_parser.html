100    def _process_slice(arg_dict: Dict[str, Any]) -> Dict[str, Any]:
101        """Execute a single processor on the received data.
102
103        Args:
104            arg_dict Dict[str, Any]: Dictionary containing anything that needs to be passed to this
105                thread for execution. Typically:
106                {
107                    'processor' Type[Processor]: Type of processor to instatiate
108                    'processor_args' Dict[str, Any]: Any arguments that should be passed to the processor

113            Dict[str, Any]: Any fields created or modified by the processors

116        processor_args = arg_dict['processor_args']
117        processor = arg_dict['processor'](**processor_args)
118        
119        #Run expensive initialisation
120        start = time.perf_counter()
121        processor.initialise()
122        arg_dict['data']['performance'][processor.name]['initialise'] = [round(time.perf_counter() - start, 3)]
123        #Run procesing
124        processor.process(arg_dict['data'])
125        
126        #Return relevant fields from output dictionary
127        return {k:arg_dict['data'][k] for k in ['metadata', 'performance'] + processor.generates()}

129    def _slice_data(self, data: Dict[str, Any], page_slices: List[List[int]], 
130                   requirements: Tuple[List[str], List[str]], processor_name: str):
131        keys = requirements[0] + [k for k in requirements[1] if k in data]
132        sliced_data = [
133            {   'metadata':data['metadata'], 
134                'performance':{processor_name:data['performance'][processor_name]}, 
135                'slice': ps
136            } | \
137            {k:{p:data[k][p] for p in ps} for k in keys}
138        for ps in page_slices]
139        return sliced_data

141    def _merge_data(self, original_data: Dict[str, Any], 
142                   sliced_data: List[Dict[str, Any]], 
143                   new_fields: List[str],
144                   processor_name: str):
145
146        for f in new_fields:
147            if f not in original_data:
148                original_data[f] = {}
149
150        perfs = original_data['performance']
151        for data_slice in sliced_data:
152            for k in new_fields:
153                original_data[k] |= data_slice[k]
154                
155            for field in data_slice['performance'][processor_name]:
156                if field not in perfs[processor_name]:
157                    perfs[processor_name][field] = []
158                perfs[processor_name][field] += data_slice['performance'][processor_name][field]
159                
160        original_data['metadata'] |= sliced_data[0]['metadata']
161                
162        return original_data

164    def _run_processor(self, processor: Type[Processor], processor_args: Dict[str, Any], 
165                      pages: List[int], data: Dict[str, Any]):
166        
167        self.logger.debug("========================= Running %s ===========================", type(processor).__name__)
168
169        start = time.perf_counter()
170        if processor.name not in data['performance']:
171            data['performance'][processor.name] = {}
172
173        if len(pages) > 1 and (not self.max_threads or self.max_threads > 1) and processor.threadable:
174            slice_size = max(self.min_slice_size, int(len(pages) / self.max_slices))
175            page_slices = [pages[i*slice_size:(i*slice_size)+slice_size] for i in range(int(len(pages)/slice_size))]
176            if len(pages) % slice_size != 0:
177                page_slices.append(pages[len(page_slices)*slice_size:])
178        else:
179            page_slices = [pages]
180
181        self.logger.debug("Page slices=%s", str(page_slices))
182        
183        proc = processor(**processor_args, log_level=self.log_level)
184        data_slices = self._slice_data(data, page_slices, proc.requirements(), processor.name)
185        thread_args = [{
186            'processor':processor, 
187            'processor_args': {
188                **processor_args, 'log_level': self.log_level
189            } ,
190            'data': data_slice
191        } for data_slice in data_slices]
192
193        if len(page_slices) > 1:
194            with mp.Pool(self.max_threads if (self.max_threads and self.max_threads > 1) else None) as p:
195                sliced_results = p.map(BurdocParser._process_slice, thread_args, chunksize=1)
196        else:
197            sliced_results = [BurdocParser._process_slice(thread_args[0])]
198
199        self._merge_data(data, sliced_results, proc.generates(), processor.name)
200        
201        data['performance'][processor.name]['total'] = round(time.perf_counter() - start, 3)

203    def _format_profile_info(self, profile_info: Dict[str, Dict[str, Union[float, List[float]]]]):
204        """Update self.last_performance with performance information
205
206        Args:
207            profile_info (Dict[str, Dict[str, Union[float, List[float]]]]): 
208                Collected processor performance info
209        """
210        perf_list = []
211        for k in profile_info:
212            perf_list.append(
213                {'name':k, 'total':profile_info[k]['total']}
214            )
215            for field in profile_info[k]:
216                if field == 'total' or field == 'name':
217                    continue
218                if isinstance(profile_info[k][field], float):
219                    perf_list[-1][field] = profile_info[k][field]
220                else:
221                    perf_list[-1][field] = round(sum(profile_info[k][field]), 3) #type:ignore
222                
223        perf_list.sort(key=lambda x: x['total'], reverse=True)
224        self.profile_info = perf_list

230        print("=================================================================")

233        if self.profile_info:
234            for entry in self.profile_info:
235                print(f"{entry['name']}:\tTotal={entry['total']}s\tInit:{entry['initialise']}s")

245    def read(self, path: str, 
246             pages: Optional[List[int]]=None, 
247             extract_images: bool=True,
248             extract_page_images: bool=False,
249             extract_page_hierarchy: bool=True,
250        ) -> Any:
251        """Read a PDF and output a structured response
252
253        Args:
254            path (str): Path of the pdf to load
255            pages (Optional[List[int]], optional): List of pages to extract. Defaults to None.
256            extract_images: (bool): Extract images from PDF. This can cause the output to become extremely large.
257                Default is False
258            extract_page_images: (bool): Extract the page images rendered as part of the processing.
259                Default is False
260            extract_page_hierarchy: Extract a list of headings and titles. Default is False.
261            extract_profile_info: Extract timing profile information for this run
262
263        Raises:
264            FileNotFoundError: If the file cannot be found. 
265            EmptyFileError: If the file has zero length. Subclass of FileDataError and RuntimeError
266            ValueError: If unknown file type is specified. Subclass of RuntimeError
267            FileDataError: If the document has an invalid structure for the given type. Subclass of RuntimeError
268
269        Returns:
270            Dict[str, Any]: Structured content, has format
271            {
272                'metadata' (Dict[str, Any]): Any metadata about the file itself
273                'content' (Dict[int, List[Any]]):  Ordered content organised per-page
274                'page_hierarchy (Dict[int, List[Any]]): Headers found in each page
275                'images', (Dict[int, List[PIL.Image.Image]], optional): Images extracted from each page.
276                    Only generated if extract_images is True
277                'page_images', (Dict[int, PIL.Image.Image], optional): Image rendered for each page.
278                    Only generated if generate_page_images is True.
279            }
280        """
281        
282        if not os.path.exists(path) or not os.path.isfile(path):
283            raise FileNotFoundError(path)
284                
285        start = time.perf_counter()
286        pdf = fitz.open(path)
287        if not pages:
288            pages = [i for i in range(pdf.page_count)]
289        else:
290            pages = [int(p) for p in pages if p < pdf.page_count]
291        pdf.close()
292
293        data = {'metadata':{'path':path}, 'performance':{'burdoc':self.performance}}
294        renderers = []
295
296        for processor, processor_args, render_processor in self.processors:
297            self._run_processor(processor, processor_args, pages, data)     
298            if render_processor:
299                renderers.append(processor(**processor_args, log_level=self.log_level))
300    
301        self.performance['total'] = round(time.perf_counter() - start, 3)
302
303        if self.show_pages:
304            render_pages(data, renderers)
305            
306        self._format_profile_info(data['performance']) #type:ignore
307        
308        return_fields = [f for f in self.default_return_fields]
309        if extract_images:
310            return_fields.append("images")
311        
312        if extract_page_images:
313            return_fields.append("page_images")
314            
315        if extract_page_hierarchy:
316            return_fields.append("page_hierarchy")
317            
318        return {k:data[k] for k in return_fields}
